exports.dialogflowFirebaseFulfillment = functions.https.onRequest((request, response) => {
  const agent = new WebhookClient({ request, response });
  console.log('Dialogflow Request headers: ' + JSON.stringify(request.headers));
  console.log('Dialogflow Request body: ' + JSON.stringify(request.body));
 
  function welcome(agent) {
    agent.add(`Welcome to my agent!`);
  }
 
  function fallback(agent) {
    agent.add(`I didn't understand`);
    agent.add(`I'm sorry, can you try again?`);
  }

  function calculator(agent) {
    var num01 = agent.parameters.num01;
    var num02 = agent.parameters.num02;
    var operator = agent.parameters.operator;
    //agent.add("Number01: "+num01+" Number02: "+num02+" Operator: "+operator);
    if (operator=="+")
    {
      var result = num01+num02;
      agent.add(`The Sum of ${num01} and ${num02} is ${result}`);
    }
    else if (operator=="-")
    {
      var result01 = num01-num02;
      agent.add(`The Subtracation of ${num01} and ${num02} is ${result01}`);
    }
    else if (operator=="/")
    {
      var result02 = num01/num02;
      agent.add(`The Division of ${num01} and ${num02} is ${result02.toFixed(2)}`);
    }
    else if (operator=="*")
    {
      var result03 = num01*num02;
      agent.add(`The Multiplication of ${num01} and ${num02} is ${result03}`);
    }
    else{
      agent.add(`You provide the wrong information`);
    }    
  }
